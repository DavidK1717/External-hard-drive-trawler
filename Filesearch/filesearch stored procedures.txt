Create PROCEDURE create_drive_record(IN @vol varchar(10), IN @total_cap int, IN @available_space int, IN @filesystem Varchar(10))
insert into drive(volume_label, total_capacity, available_space, file_system) values (vol, total_cap, available_space, filesystem)


Create PROCEDURE update_drive_record(IN @vol varchar(10), IN @total_cap int, IN @available_space int, IN @filesystem Varchar(10))
BEGIN
update drive
set
total_capacity=@total_cap, available_space = @available_space, filesystem = @filesystem
where vol = @vol;
END

Create PROCEDURE drive_check(IN @vol varchar(10))
select count(*) from drive where volume_label = @vol


Create PROCEDURE create_file_record(IN @volume varchar(10), IN @path Varchar(1000), IN @file_name Varchar(1000), IN @file_ext Varchar(3), IN @size int)
insert into file(volume_label, filepath, filename, file_ex) values (@volume, @path, @file_name, @file_ext)
